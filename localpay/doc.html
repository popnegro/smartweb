Script de Integración Express

Utiliza Node.js para recibir la notificación de pago y guardar el estado en la base de datos de forma ultra-rápida.

const express = require('express');
const app = express();
app.use(express.json());

// Endpoint para recibir Webhooks de Mercado Pago
app.post('/webhook', async (req, res) => {
    const { data, type } = req.body;

    // Solo procesamos pagos
    if (type === 'payment') {
        const paymentId = data.id;
        
        // 1. Consultar a MP el estado real del pago
        const response = await fetch(`https://api.mercadopago.com/v1/payments/${paymentId}`, {
            headers: { 'Authorization': `Bearer ${process.env.MP_ACCESS_TOKEN}` }
        });
        const paymentData = await response.json();

        // 2. Si está aprobado, actualizar Semáforo en la DB
        if (paymentData.status === 'approved') {
            await supabase.from('payments').update({ status: 'VERDE' }).eq('id', paymentId);
        }
    }
    res.sendStatus(200);
});

app.listen(3000, () => console.log('Server LocalPay Activo'));

Mecanismo de Sincronización (Polling de Alta Frecuencia)

En la versión Low Cost, para evitar servidores de WebSockets costosos, implementamos un fetch optimizado cada 2.5 segundos:

const checkStatus = setInterval(async () => {
  const { data, error } = await supabase
    .from('payments')
    .select('status')
    .eq('order_id', currentOrder)
    .single();

  if (data.status === 'VERDE') {
    triggerSuccessUI(); // Cambia Semáforo a Verde
    clearInterval(checkStatus);
  }
}, 2500);
Seguridad Anti-Spoofing: El cambio a "Aprobado" solo ocurre si el servidor recibe un Webhook firmado por Mercado Pago. El Frontend no tiene permisos para cambiar su propio estado manualmente.
